def self.included base
  base.class_eval do
    # don't actually need most of these
    layout :parent, GObject::GObject::Struct,
      :constructed, :int,
      :static_object, :int,
      :argument_table, :pointer,
      :nickname, :string,
      :description, :string,
      :preclose, :int,
      :close, :int,
      :postclose, :int,
      :local_memory, :size_t
  end
end
